"use client";

import { useEffect, useState } from "react";
import { useParams } from "next/navigation";
import { db } from "@/lib/firebase";
import { doc, getDoc } from "firebase/firestore";
import {
  Calendar,
  Gauge,
  Car,
  MapPin,
  Euro,
  Info,
} from "lucide-react";

import CarGallery from "@/components/listings/CarGallery"; // üß© Import galeria ta

export const dynamic = "force-dynamic";

/* üß† Helper universal pentru citirea valorilor din schemaData */
function findValue(schemaData: any, key: string) {
  if (!schemaData || typeof schemaData !== "object") return undefined;
  const normalized = key.trim().toLowerCase();

  for (const [sectionName, sectionData] of Object.entries(schemaData)) {
    if (typeof sectionData === "object" && sectionData !== null) {
      for (const [k, v] of Object.entries(sectionData)) {
        if (k.trim().toLowerCase() === normalized) return v;
      }
    }
  }

  return undefined;
}

export default function CarDetailsPage() {

  const [car, setCar] = useState<any>(null);

      if (!id) return;
      try {
        const docRef = doc(db, "cars", id as string);
        const snap = await getDoc(docRef);
        if (snap.exists()) {
          const carData = { id, ...snap.data() };
          setCar(carData);
          console.log("[CarDetails] Car loaded:", carData);
        } else {
          console.warn("[CarDetails] Car not found for id:", id);
        }
